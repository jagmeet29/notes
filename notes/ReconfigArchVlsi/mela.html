Melay code for 101 Overlaping

    
module melay(x,clk,out);
input x,clk;
output reg out=0;
reg [1:0]state=2'b00;
always @(posedge clk)
case(state)
2'b00:begin
if(x==1)
begin
state=2'b01;
out=0;
end
end
2'b01:begin
if(x==0)
begin
state=2'b10;
out=0;
end
end
2'b10:begin
if(x==1)
begin
state=2'b01;
out=1;
end
else
begin
state=2'b00;
out=0;
end
end
endcase
endmodule
